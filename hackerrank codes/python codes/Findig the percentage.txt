if __name__ == '__main__':
    n = int(input())
    student_marks = {}
    for _ in range(n):
        name, *line = input().split()
        scores = list(map(float, line))
        student_marks[name] = scores
    query_name = input()

#print(student_marks)
#print(name)
for key in student_marks:
    if key == query_name:
        a = student_marks[key]
        avg = sum(a)/3
        print('%.2f'%avg)
